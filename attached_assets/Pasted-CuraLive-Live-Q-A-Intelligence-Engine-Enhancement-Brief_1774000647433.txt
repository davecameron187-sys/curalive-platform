      sessionId,
      platform,
      shareLink: link,
      createdAt: new Date()
    });

    return link;
  }

  // Generate embed code for platforms
  static getEmbedCode(sessionId: string, whiteLabel: boolean = false) {
    const theme = whiteLabel ? '&theme=platform' : '';
    return `<iframe 
      src="https://curalive.app/embed/qa/${sessionId}${theme}" 
      width="380" 
      height="620" 
      frameborder="0" 
      allowfullscreen>
    </iframe>`;
  }

  // Post-event intelligence report for platform admins
  static async sendPlatformReport(sessionId: string, platform: string) {
    const report = await this.generateEventSummary(sessionId);
    this.ably.publish(`platform-${platform}-report`, { sessionId, report });
  }
}
2. client/components/PlatformShareButton.tsx (One-Click Button for Zoom/Teams Chat)
tsximport { trpc } from '../trpc';
import { Button } from '@/components/ui/button';
import { Share2 } from 'lucide-react';

export function PlatformShareButton({ sessionId, platform }: { sessionId: string; platform: string }) {
  const generateLink = trpc.platformEmbed.generateShareLink.useMutation();

  const handleShare = async () => {
    const link = await generateLink.mutateAsync({ sessionId, platform });
    navigator.clipboard.writeText(`Ask smart questions with CuraLive AI → ${link}`);
    alert('✅ Smart Q&A link copied! Paste into Zoom/Teams chat.');
  };

  return (
    <Button onClick={handleShare} className="flex items-center gap-2 bg-gradient-to-r from-blue-600 to-purple-600">
      <Share2 size={18} />
      Open Smart Q&A with CuraLive
    </Button>
  );
}

tsximport { useState, useEffect } from 'react';
import { trpc } from '../trpc';
import { Input } from '@/components/ui/input';
import { Button } from '@/components/ui/button';

export function EmbeddableQaWidget({ sessionId, whiteLabel = false }: { sessionId: string; whiteLabel?: boolean }) {
  const [questions, setQuestions] = useState([]);
  const [newQuestion, setNewQuestion] = useState('');
  const submitQuestion = trpc.liveQa.submitQuestion.useMutation();

  const sendQuestion = () => {
    submitQuestion.mutate({ sessionId, question: newQuestion });
    setNewQuestion('');
  };

  // Real-time Ably subscription (already in your stack)
  useEffect(() => {
    // Subscribe to event-${sessionId}-qa
  }, [sessionId]);

  return (
    <div className={`w-full max-w-[380px] h-[620px] border rounded-xl overflow-hidden ${whiteLabel ? 'border-gray-300' : 'border-purple-500'}`}>
      <div className="bg-gradient-to-r from-blue-600 to-purple-600 p-4 text-white font-semibold text-center">
        {whiteLabel ? 'Smart Q&A' : 'CuraLive Smart Q&A'}
      </div>
      
      {/* Question list */}
      <div className="h-[420px] overflow-auto p-4 space-y-3">
        {questions.map((q: any) => (
          <div key={q.id} className="bg-white p-3 rounded-lg shadow-sm">
            <p className="text-sm">{q.question}</p>
            {q.status === 'answered' && <p className="text-xs text-green-600 mt-1">Answered ✓</p>}
          </div>
        ))}
      </div>

      {/* Input */}
      <div className="p-4 border-t bg-white">
        <div className="flex gap-2">
          <Input 
            value={newQuestion} 
            onChange={(e) => setNewQuestion(e.target.value)}
            placeholder="Type or speak your question..."
            onKeyDown={(e) => e.key === 'Enter' && sendQuestion()}
          />
          <Button onClick={sendQuestion}>Send</Button>
        </div>
        <p className="text-[10px] text-gray-500 mt-2 text-center">
          Powered by CuraLive AI • Zero login required
        </p>
      </div>
    </div>
  );
}
