TypeScript// @ts-nocheck
import { invokeLLM } from '../utils/invokeLLM';
import { AblyRealtimeService } from './AblyRealtimeService';
import { getDb } from '../db';

export class AgiComplianceService {
  private ably = AblyRealtimeService.getInstance();

  static async assessQuestion(question: string, companyId: string, context: any) {
    const prompt = `
      Analyse this question for compliance risk across JSE, Reg FD (US), MAR (EU), POPIA/GDPR, King IV.
      Company context: ${JSON.stringify(context)}
      Return STRICT JSON only:
      {
        riskScore: number (0-1),
        jurisdictionFlags: string[],
        autoDraft: string,
        policySuggestion: object | null,
        recommendedAction: "forward" | "route_to_bot" | "legal_review" | "delay_24h"
      }
    `;

    const result = await invokeLLM(prompt, { temperature: 0.1, responseFormat: 'json' });

    if (result.riskScore > 0.7) {
      await this.ably.publish(`company-${companyId}-compliance-alert`, {
        type: 'high_risk',
        question,
        ...result
      });
      // Auto-promote new policy via Module 28
      if (result.policySuggestion) {
        await getDb().insert(compliancePolicies).values({
          companyId,
          policy: result.policySuggestion,
          source: 'agi_generator'
        });
      }
    }

    return result;
  }
}
